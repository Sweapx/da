import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

sns.set_theme(style="whitegrid")

# ==============================================================================
# 1. KONFIGURASI AWAL
# ==============================================================================
NAMA_FILE = 'dataset_raw.csv'

# GUNAKAN KURUNG SIKU [] JIKA LEBIH DARI SATU KOLOM
# Contoh 1 kolom  : KOLOM_TARGET = ['Sales']
# Contoh banyak   : KOLOM_TARGET = ['Sales', 'Profit', 'Quantity']
KOLOM_TARGET = ['Sales', 'Profit']

print("Mulai proses data...")

# ==============================================================================
# 2. LOAD DATA
# ==============================================================================
df = pd.read_csv(NAMA_FILE)

# ==============================================================================
# 3. TELAAH DATA (DATA UNDERSTANDING)
# ==============================================================================
print("\n--- TELAAH DATA ---")
print(f"Shape: {df.shape}")
print("\nInfo Data:")
df.info()
print("\nStatistik Numerik:")
display(df.describe())

# ==============================================================================
# 4. PERBAIKAN TIPE DATA (JIKA ADA YANG SALAH)
# ==============================================================================
# if 'Postal Code' in df.columns:
#     df['Postal Code'] = df['Postal Code'].astype(str)
#     print("\n[INFO] Tipe data 'Postal Code' diubah menjadi teks.")

kolom_numerik = df.select_dtypes(include=[np.number]).columns.tolist()

# ==============================================================================
# 5. HAPUS DUPLIKAT
# ==============================================================================
print("\n--- PEMBERSIHAN DATA ---")
duplikat = df.duplicated().sum()
if duplikat > 0:
    df.drop_duplicates(inplace=True)
print(f"Duplikat dihapus: {duplikat}")

# ==============================================================================
# 6. ISI MISSING VALUES
# ==============================================================================
ada_missing = False
for col in df.columns:
    kosong = df[col].isnull().sum()
    if kosong > 0:
        ada_missing = True
        if col in kolom_numerik:
            df[col] = df[col].fillna(df[col].median())
            print(f"Missing value '{col}' diisi median.")
        else:
            df[col] = df[col].fillna("Unknown")
            print(f"Missing value '{col}' diisi 'Unknown'.")

if not ada_missing:
    print("Aman, tidak ada missing value.")

# ==============================================================================
# 7. OUTLIER (MULTI-KOLOM), TAMPILKAN, & SIMPAN
# ==============================================================================
kondisi_outlier = pd.Series(False, index=df.index)
kolom_diproses = []

print("\n--- HASIL OUTLIER ---")
for col in KOLOM_TARGET:
    if col in df.columns:
        kolom_diproses.append(col)
        Q1         = df[col].quantile(0.25)
        Q3         = df[col].quantile(0.75)
        IQR        = Q3 - Q1
        batas_atas = Q3 + (1.5 * IQR)

        kondisi_outlier = kondisi_outlier | (df[col] > batas_atas)

        print(f"Batas atas '{col}': {batas_atas}")

        plt.figure(figsize=(8, 2))
        sns.boxplot(x=df[col], color='skyblue')
        plt.title(f'Boxplot {col}')
        plt.show()

if len(kolom_diproses) > 0:
    df_reguler = df[~kondisi_outlier].copy()
    df_vip     = df[kondisi_outlier].copy()

    print(f"\nTotal Data Reguler : {df_reguler.shape[0]} baris")
    print(f"Total Data VIP     : {df_vip.shape[0]} baris")

    print("\n--- TAMPILAN DATA BERSIH (REGULER) ---")
    display(df_reguler.head())

    df_reguler.to_csv(f"CLEAN_REG_{NAMA_FILE}", index=False)
    df_vip.to_csv(f"CLEAN_VIP_{NAMA_FILE}", index=False)
    print("\nFile reguler & VIP berhasil disimpan.")

else:
    print("\nKolom target tidak ditemukan. Outlier dilewati.")
    print("\n--- TAMPILAN DATA BERSIH ---")
    display(df.head())

    df.to_csv(f"CLEAN_{NAMA_FILE}", index=False)
    print("\nFile bersih berhasil disimpan.")
